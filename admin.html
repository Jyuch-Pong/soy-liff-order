// supabase/functions/admin-api/index.ts // // Admin 後台 API v2(以 ADMIN_PASSWORD 密鑰保護,每個請求需帶 x-admin-key header) // // v2 新增: // 刪除店家/商品(有訂單記錄者擋下,提示改用停用) // 司機管理(新增/編輯/刪除司機、帳號密碼、車號、路線指派) // 路線管理(依route_code+day_group彙整店家) import { createClient } from "npm:@supabase/supabase-js@2"; import { createHash } from "node:crypto"; const ADMIN_PASSWORD = Deno.env.get("ADMIN_PASSWORD")!; const supabase = createClient( Deno.env.get("SUPABASE_URL")!, Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!, ); const cors = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "GET, POST, OPTIONS", "Access-Control-Allow-Headers": "Content-Type, x-admin-key", }; function json(data: unknown, status = 200) { return new Response(JSON.stringify(data), { status, headers: { ...cors, "Content-Type": "application/json" }, }); } function hashPassword(pw: string): string { return createHash("sha256").update(pw).digest("hex"); } Deno.serve(async (req) => { if (req.method === "OPTIONS") return new Response("ok", { headers: cors }); if (req.headers.get("x-admin-key") !== ADMIN_PASSWORD) { return json({ error: "密碼錯誤" }, 401); } const url = new URL(req.url); const action = url.searchParams.get("action") ?? ""; try { // ================= GET ================= if (req.method === "GET") { if (action === "stores") { const { data } = await supabase.from("stores").select("*").order("store_code"); return json(data ?? []); } if (action === "products") { const { data } = await supabase.from("products").select("*").order("name"); return json(data ?? []); } if (action === "custom_prices") { const storeId = url.searchParams.get("storeId")!; const { data } = await supabase.from("store_product_prices") .select("product_id, custom_price").eq("store_id", storeId); return json(data ?? []); } if (action === "orders") { const date = url.searchParams.get("date"); let q = supabase.from("orders") .select("id, delivery_date, order_date, status, source, total_amount, created_at, stores(store_code, name), order_items(quantity, unit_price, products(name, unit))") .order("created_at", { ascending: false }) .limit(200); if (date) q = q.eq("delivery_date", date); const { data } = await q; return json(data ?? []); } if (action === "summary") { const date = url.searchParams.get("date")!; const { data } = await supabase.from("orders") .select("id, status, order_items(quantity, product_id, products(name, unit))") .eq("delivery_date", date).neq("status", "cancelled"); const totals: Record = {}; for (const o of data ?? []) { for (const it of (o as any).order_items ?? []) { const pid = it.product_id; if (!totals[pid]) totals[pid] = { name: it.products.name, unit: it.products.unit, qty: 0, storeCount: 0 }; totals[pid].qty += Number(it.quantity); totals[pid].storeCount += 1; } } return json(Object.values(totals)); } if (action === "drivers") { const { data: drivers } = await supabase.from("drivers") .select("id, name, phone, car_no, account, status").order("name"); const { data: routes } = await supabase.from("driver_routes") .select("id, driver_id, route_code, day_group"); const list = (drivers ?? []).map((d) => ({ ...d, routes: (routes ?? []).filter((r) => r.driver_id === d.id), })); return json(list); } if (action === "routes_overview") { const { data: stores } = await supabase.from("stores") .select("store_code, name, route_code, day_group, status") .order("store_code"); const { data: driverRoutes } = await supabase.from("driver_routes") .select("route_code, day_group, drivers(name, car_no)"); // 依 route_code → day_group 分組 const routes: Record> = {}; for (const s of stores ?? []) { const rc = s.route_code || "未分配"; const dg = s.day_group || "未設定"; if (!routes[rc]) routes[rc] = {}; if (!routes[rc][dg]) routes[rc][dg] = []; routes[rc][dg].push({ code: s.store_code, name: s.name, status: s.status }); } return json({ routes, driverRoutes: driverRoutes ?? [] }); } return json({ error: "unknown action" }, 400); } // ================= POST ================= if (req.method === "POST") { const body = await req.json(); // ---- 店家 ---- if (action === "create_store") { const { store_code, name, address, phone, route_code, day_group } = body; if (!store_code || !name) return json({ error: "店號與店名必填" }, 400); const { error } = await supabase.from("stores").insert({ store_code, name, address, phone, route_code, day_group, status: "active", }); if (error) return json({ error: error.message }, 500); return json({ ok: true }); } if (action === "update_store") { const { id, ...fields } = body; const { error } = await supabase.from("stores").update(fields).eq("id", id); if (error) return json({ error: error.message }, 500); return json({ ok: true }); } if (action === "delete_store") { const { id } = body; // 有訂單記錄的店家不允許真刪除 const { count } = await supabase.from("orders") .select("id", { count: "exact", head: true }).eq("store_id", id); if ((count ?? 0) > 0) { return json({ error: "此店家已有訂單記錄,無法刪除(會破壞歷史帳)。請改用「停用」。" }, 400); } await supabase.from("store_product_prices").delete().eq("store_id", id); const { error } = await supabase.from("stores").delete().eq("id", id); if (error) return json({ error: error.message }, 500); return json({ ok: true }); } // ---- 商品 ---- if (action === "create_product") { const { name, unit, base_price } = body; if (!name) return json({ error: "商品名稱必填" }, 400); const { error } = await supabase.from("products").insert({ name, unit: unit || "份", base_price: Number(base_price) || 0, }); if (error) return json({ error: error.message }, 500); return json({ ok: true }); } if (action === "update_product") { const { id, ...fields } = body; if (fields.base_price != null) fields.base_price = Number(fields.base_price); const { error } = await supabase.from("products").update(fields).eq("id", id); if (error) return json({ error: error.message }, 500); return json({ ok: true }); } if (action === "delete_product") { const { id } = body; // 有訂單記錄的商品不允許真刪除 const { count } = await supabase.from("order_items") .select("id", { count: "exact", head: true }).eq("product_id", id); if ((count ?? 0) > 0) { return json({ error: "此商品已有訂單記錄,無法刪除(會破壞歷史帳)。請改用「下架」。" }, 400); } await supabase.from("store_product_prices").delete().eq("product_id", id); const { error } = await supabase.from("products").delete().eq("id", id); if (error) return json({ error: error.message }, 500); return json({ ok: true }); } // ---- 客製價 ---- if (action === "set_custom_price") { const { storeId, productId, price } = body; if (price === "" || price == null) { await supabase.from("store_product_prices") .delete().eq("store_id", storeId).eq("product_id", productId); } else { await supabase.from("store_product_prices").upsert( { store_id: storeId, product_id: productId, custom_price: Number(price) }, { onConflict: "store_id,product_id" }, ); } return json({ ok: true }); } // ---- 訂單 ---- if (action === "cancel_order") { const { orderId, reason } = body; const { data: before } = await supabase.from("orders").select("*").eq("id", orderId).single(); const { error } = await supabase.from("orders") .update({ status: "cancelled" }).eq("id", orderId); if (error) return json({ error: error.message }, 500); await supabase.from("order_edit_logs").insert({ order_id: orderId, edited_by: "admin", edit_type: "cancel", before_data: before, after_data: { ...before, status: "cancelled" }, reason: reason || "(未填寫)", }); return json({ ok: true }); } // ---- 司機 ---- if (action === "create_driver") { const { name, phone, car_no, account, password } = body; if (!name || !account || !password) return json({ error: "姓名、帳號、密碼必填" }, 400); const { error } = await supabase.from("drivers").insert({ name, phone, car_no, account, password_hash: hashPassword(password), status: "active", }); if (error) return json({ error: error.message }, 500); return json({ ok: true }); } if (action === "update_driver") { const { id, password, ...fields } = body; if (password) fields.password_hash = hashPassword(password); const { error } = await supabase.from("drivers").update(fields).eq("id", id); if (error) return json({ error: error.message }, 500); return json({ ok: true }); } if (action === "delete_driver") { const { id } = body; await supabase.from("driver_routes").delete().eq("driver_id", id); const { error } = await supabase.from("drivers").delete().eq("id", id); if (error) return json({ error: error.message }, 500); return json({ ok: true }); } if (action === "add_driver_route") { const { driverId, route_code, day_group } = body; if (!driverId || !route_code) return json({ error: "參數不完整" }, 400); const { error } = await supabase.from("driver_routes").insert({ driver_id: driverId, route_code, day_group: day_group || null, }); if (error) return json({ error: error.message }, 500); return json({ ok: true }); } if (action === "delete_driver_route") { const { id } = body; const { error } = await supabase.from("driver_routes").delete().eq("id", id); if (error) return json({ error: error.message }, 500); return json({ ok: true }); } return json({ error: "unknown action" }, 400); } return json({ error: "method not allowed" }, 405); } catch (e) { return json({ error: String(e) }, 500); } });
