// supabase/functions/driver-api/index.ts // // 司機端 API: // POST ?action=login → 驗證帳號密碼,回傳司機資訊 // GET ?action=today&date=YYYY-MM-DD → 當日配送清單(綠/紅)+品項總計 // POST ?action=create_onsite → 現場快速建單(當日配送) // POST ?action=addon → 現場追加(合併進既有訂單) // POST ?action=mark_delivered → 送達回報 // // 驗證方式:每個請求帶 x-driver-account / x-driver-password header, // 伺服器端以 SHA-256 比對 drivers.password_hash import { createClient } from "npm:@supabase/supabase-js@2"; import { createHash } from "node:crypto"; const supabase = createClient( Deno.env.get("SUPABASE_URL")!, Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!, ); const cors = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "GET, POST, OPTIONS", "Access-Control-Allow-Headers": "Content-Type, x-driver-account, x-driver-password", }; function json(data: unknown, status = 200) { return new Response(JSON.stringify(data), { status, headers: { ...cors, "Content-Type": "application/json" }, }); } function hashPassword(pw: string): string { return createHash("sha256").update(pw).digest("hex"); } // 台灣時區的 YYYY-MM-DD function taipeiDateStr(offsetDays = 0): string { const now = new Date(Date.now() + offsetDays * 86400000); return now.toLocaleDateString("en-CA", { timeZone: "Asia/Taipei" }); } // 日期 → day_group(一四/二五/三六),週日回傳 null function dayGroupOf(dateStr: string): string | null { const wd = new Date(dateStr + "T00:00:00+08:00").getUTCDay(); // 用+08:00確保取到台灣的星期幾 // getUTCDay對+08:00午夜:直接用Date物件的getDay在UTC環境會偏移,改用可靠算法: const d = new Date(dateStr + "T12:00:00+08:00"); const weekday = new Intl.DateTimeFormat("en-US", { timeZone: "Asia/Taipei", weekday: "short" }).format(d); const map: Record = { Mon: "一四", Thu: "一四", Tue: "二五", Fri: "二五", Wed: "三六", Sat: "三六", Sun: null, }; return map[weekday] ?? null; } // 驗證司機身份,成功回傳司機資料,失敗回傳null async function authDriver(req: Request) { const account = req.headers.get("x-driver-account") ?? ""; const password = req.headers.get("x-driver-password") ?? ""; if (!account || !password) return null; const { data: driver } = await supabase.from("drivers") .select("id, name, car_no, account, password_hash, status") .eq("account", account).eq("status", "active").maybeSingle(); if (!driver) return null; if (driver.password_hash !== hashPassword(password)) return null; return driver; } // 取得某店某日的價格表(客製價優先) async function getPriceMap(storeId: string) { const { data: products } = await supabase.from("products") .select("id, name, unit, base_price").eq("is_active", true); const { data: custom } = await supabase.from("store_product_prices") .select("product_id, custom_price").eq("store_id", storeId); const customMap = new Map((custom ?? []).map((c) => [c.product_id, Number(c.custom_price)])); const map = new Map(); for (const p of products ?? []) { map.set(p.id, { name: p.name, unit: p.unit, price: customMap.has(p.id) ? customMap.get(p.id)! : Number(p.base_price), }); } return map; } Deno.serve(async (req) => { if (req.method === "OPTIONS") return new Response("ok", { headers: cors }); const url = new URL(req.url); const action = url.searchParams.get("action") ?? ""; try { // ---- 登入 ---- if (req.method === "POST" && action === "login") { const driver = await authDriver(req); if (!driver) return json({ error: "帳號或密碼錯誤" }, 401); return json({ ok: true, driver: { id: driver.id, name: driver.name, car_no: driver.car_no } }); } // 以下所有 action 都需要驗證 const driver = await authDriver(req); if (!driver) return json({ error: "請重新登入" }, 401); // ---- 當日清單 ---- if (req.method === "GET" && action === "today") { const date = url.searchParams.get("date") || taipeiDateStr(0); const dg = dayGroupOf(date); // 司機負責的路線 const { data: myRoutes } = await supabase.from("driver_routes") .select("route_code, day_group").eq("driver_id", driver.id); // 今天生效的路線:day_group=null(全包) 或 day_group=今天的梯次 const activeRouteCodes = (myRoutes ?? []) .filter((r) => r.day_group === null || r.day_group === dg) .map((r) => r.route_code); if (!dg || activeRouteCodes.length === 0) { return json({ date, dayGroup: dg, routes: activeRouteCodes, stores: [], totals: [] }); } // 該路線+梯次的店家 const { data: stores } = await supabase.from("stores") .select("id, store_code, name, address, phone") .in("route_code", activeRouteCodes) .eq("day_group", dg) .eq("status", "active") .order("store_code"); const storeIds = (stores ?? []).map((s) => s.id); // 該日訂單 const { data: orders } = storeIds.length ? await supabase.from("orders") .select("id, store_id, status, source, total_amount, order_items(product_id, quantity, unit_price, products(name, unit))") .in("store_id", storeIds) .eq("delivery_date", date) .neq("status", "cancelled") : { data: [] }; const orderByStore = new Map((orders ?? []).map((o) => [o.store_id, o])); // 品項總計 const totals: Record = {}; for (const o of orders ?? []) { for (const it of (o as any).order_items ?? []) { const pid = it.product_id; if (!totals[pid]) totals[pid] = { name: it.products.name, unit: it.products.unit, qty: 0, storeCount: 0 }; totals[pid].qty += Number(it.quantity); totals[pid].storeCount += 1; } } const list = (stores ?? []).map((s) => { const o = orderByStore.get(s.id) as any; return { id: s.id, store_code: s.store_code, name: s.name, address: s.address, order: o ? { id: o.id, status: o.status, source: o.source, total: Number(o.total_amount), items: (o.order_items ?? []).map((it: any) => ({ productId: it.product_id, name: it.products.name, qty: Number(it.quantity), price: Number(it.unit_price), })), } : null, }; }); // 商品清單(給現場建單/追加選品用,不含價格——價格依店家於送出時後端計算) const { data: products } = await supabase.from("products") .select("id, name, unit").eq("is_active", true).order("name"); return json({ date, dayGroup: dg, routes: activeRouteCodes, stores: list, totals: Object.values(totals), products: products ?? [] }); } // ---- 現場快速建單 ---- if (req.method === "POST" && action === "create_onsite") { const { storeId, date, items } = await req.json(); if (!storeId || !date || !Array.isArray(items) || items.length === 0) { return json({ error: "參數不完整" }, 400); } const priceMap = await getPriceMap(storeId); const rows = items .filter((it: any) => priceMap.has(it.productId) && Number(it.qty) > 0) .map((it: any) => ({ product_id: it.productId, quantity: Number(it.qty), unit_price: priceMap.get(it.productId)!.price, })); if (rows.length === 0) return json({ error: "沒有有效品項" }, 400); const total = rows.reduce((s, r) => s + r.quantity * r.unit_price, 0); const { data: order, error } = await supabase.from("orders").insert({ store_id: storeId, order_date: taipeiDateStr(0), delivery_date: date, status: "confirmed", source: "on_site", total_amount: total, created_by: driver.name, }).select("id").single(); if (error) return json({ error: error.message }, 500); await supabase.from("order_items").insert(rows.map((r) => ({ ...r, order_id: order.id }))); return json({ ok: true }); } // ---- 現場追加 ---- if (req.method === "POST" && action === "addon") { const { orderId, storeId, items } = await req.json(); if (!orderId || !storeId || !Array.isArray(items) || items.length === 0) { return json({ error: "參數不完整" }, 400); } const priceMap = await getPriceMap(storeId); const { data: existing } = await supabase.from("order_items") .select("id, product_id, quantity").eq("order_id", orderId); const existMap = new Map((existing ?? []).map((e) => [e.product_id, e])); let addedTotal = 0; for (const it of items) { const qty = Number(it.qty); if (!priceMap.has(it.productId) || qty <= 0) continue; const price = priceMap.get(it.productId)!.price; addedTotal += qty * price; const ex = existMap.get(it.productId); if (ex) { await supabase.from("order_items") .update({ quantity: Number(ex.quantity) + qty }).eq("id", ex.id); } else { await supabase.from("order_items").insert({ order_id: orderId, product_id: it.productId, quantity: qty, unit_price: price, }); } } if (addedTotal === 0) return json({ error: "沒有有效品項" }, 400); // 更新訂單總額 const { data: order } = await supabase.from("orders") .select("total_amount").eq("id", orderId).single(); await supabase.from("orders") .update({ total_amount: Number(order!.total_amount) + addedTotal }).eq("id", orderId); return json({ ok: true }); } // ---- 送達回報 ---- if (req.method === "POST" && action === "mark_delivered") { const { orderId } = await req.json(); const { error } = await supabase.from("orders") .update({ status: "delivered" }).eq("id", orderId); if (error) return json({ error: error.message }, 500); return json({ ok: true }); } return json({ error: "unknown action" }, 400); } catch (e) { return json({ error: String(e) }, 500); } });
